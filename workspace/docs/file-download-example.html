// ===== 1. 下载文件 =====
async function downloadFile(filename) {
  const token = document.getElementById('token-input').value;
  const gateway = document.getElementById('gateway-url').value;
  
  const response = await fetch(
    `${gateway}/__openclaw__/canvas/${encodeURIComponent(filename)}`,
    {
      headers: { 'Authorization': `Bearer ${token}` },
      // HTTPS 自签名证书需要忽略（开发环境）
      // 生产环境请使用有效证书
      mode: 'cors'
    }
  );
  
  if (!response.ok) {
    throw new Error(`下载失败: ${response.status}`);
  }
  
  return await response.blob();
}

// ===== 2. WSS 订阅会话消息 =====
function connectWSS() {
  const gateway = document.getElementById('gateway-url').value.replace('https', 'wss');
  const token = document.getElementById('token-input').value;
  
  const ws = new WebSocket(gateway);
  
  ws.onopen = () => {
    // 1. 发送 connect 请求
    ws.send(JSON.stringify({
      type: 'req',
      id: '1',
      method: 'connect',
      params: {
        minProtocol: 3,
        maxProtocol: 3,
        client: { id: 'web', version: '1.0', platform: 'web', mode: 'operator' },
        role: 'operator',
        scopes: ['operator.read', 'operator.write'],
        auth: { token: token },
        locale: 'zh-CN',
        userAgent: 'file-download-demo/1.0'
      }
    }));
  };
  
  ws.onmessage = (event) => {
    const msg = JSON.parse(event.data);
    console.log('收到消息:', msg);
    
    // 解析 session.message 事件
    if (msg.event === 'session.message') {
      const messages = msg.payload?.messages || [];
      messages.forEach(m => {
        // 检查工具调用结果中的文件路径
        if (m.role === 'toolResult' && m.toolName === 'write') {
          console.log('文件已生成:', m.content);
          // m.content 通常包含 "文件已写入: /path/to/file.txt"
          const match = m.content.match(/写入:\s*(.+)/);
          if (match) {
            const filepath = match[1];
            const filename = filepath.split('/').pop();
            console.log('文件名:', filename);
            // 自动下载
            downloadFile(filename);
          }
        }
      });
    }
  };
  
  // 2. 连接成功后，订阅消息
  ws.on('open', () => {
    // 发送订阅请求
    ws.send(JSON.stringify({
      type: 'req',
      id: '2',
      method: 'sessions.messages.subscribe',
      params: { key: 'agent:main:telegram:direct:123' } // 替换为你的 sessionKey
    }));
  });
}
